    <script>
        const lenis = new Lenis({ duration: 1.5 });
        function raf(time) { lenis.raf(time); requestAnimationFrame(raf); }
        requestAnimationFrame(raf);

        const canvas = document.getElementById('dna-canvas');
        const ctx = canvas.getContext('2d');
        let particles = [];
        function init() {
            canvas.width = window.innerWidth; canvas.height = window.innerHeight;
            particles = [];
            for (let i = 0; i < 80; i++) particles.push({ x: Math.random() * canvas.width, y: Math.random() * canvas.height, sx: (Math.random()-0.5)*0.2, sy: (Math.random()-0.5)*0.2 });
        }
        function anim() {
            ctx.clearRect(0,0,canvas.width,canvas.height);
            particles.forEach((p, i) => {
                p.x += p.sx; p.y += p.sy;
                if (p.x > canvas.width) p.x=0; if (p.y > canvas.height) p.y=0;
                particles.slice(i+1).forEach(p2 => {
                    let d = Math.hypot(p.x-p2.x, p.y-p2.y);
                    if (d < 120) { ctx.strokeStyle = `rgba(180,210,255,${0.15*(1-d/120)})`; ctx.beginPath(); ctx.moveTo(p.x,p.y); ctx.lineTo(p2.x,p2.y); ctx.stroke(); }
                });
            });
            requestAnimationFrame(anim);
        }
        init(); anim();

        // 4 CATEGORIES ROTATION + ARROWS
        let currentSlide = 0;
        const slides = document.querySelectorAll('.cat-slide');
        
        function showSlide(index) {
            slides[currentSlide].classList.remove('active');
            currentSlide = (index + slides.length) % slides.length;
            slides[currentSlide].classList.add('active');
        }

        function moveSlide(step) {
            clearInterval(autoRotate); // Спира авто-въртенето при клик
            showSlide(currentSlide + step);
        }

        let autoRotate = setInterval(() => {
            showSlide(currentSlide + 1);
        }, 5000);

        const observer = new IntersectionObserver(entries => {
            entries.forEach(entry => { if(entry.isIntersecting) entry.target.classList.add('active'); });
        }, { threshold: 0.1 });
        document.querySelectorAll('section').forEach(s => observer.observe(s));
    </script>
